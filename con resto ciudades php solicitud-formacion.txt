    $query = "SELECT t.documento_tercero, t.nombre_completo, d.depto_nom_propio, f.nombre_fac_min, t.CARGO_ADMIN

              LEFT JOIN deparmanentos d ON t.fk_depto = d.PK_DEPTO

              WHERE t.documento_tercero = ?";
